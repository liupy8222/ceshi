# 构建单年图函数（返回 g_final grob）
# =======================================================
build_year_grob <- function(df_year, year_label, show_xaxis = FALSE) {
  
  p <- ggplot(df_year, aes(x = Date)) +
    
    geom_rect(aes(xmin = Date - 0.4, xmax = Date + 0.4,
                  ymin = ylim_min,
                  ymax = ylim_min + (Precip_Sum * coeff_precip),
                  fill = "降雨量"),
              alpha = 0.8) +
    
    geom_ribbon(aes(ymin = Temp_Min, ymax = Temp_Max, fill = "日温度范围"),
                alpha = 0.4, color = "#4D7668", size = 0.2) +
    
    geom_line(aes(y = Temp_Avg, color = "日平均温度"), linewidth = 0.4) +
    
    geom_line(aes(y = ylim_min + (Humidity_Avg * coeff_hum), color = "相对湿度"),
              linewidth = 0.4) +
    
    # 年份标注（右上角）
    annotate("text", x = max(df_year$Date), y = ylim_max - 1,
             label = as.character(year_label),
             hjust = 1, vjust = 1, size = 3.5, fontface = "bold",
             family = "SimHei") +
    
    scale_fill_manual(name = "",
                      values = c("日温度范围" = "#C5D9E2",
                                 "降雨量" = "#4CACD6")) +
    scale_color_manual(name = "",
                       values = c("日平均温度" = "#F08A8A",
                                  "相对湿度" = "#F5B955")) +
    
    scale_y_continuous(
      name = expression(bold("温度 ("*degree*"C)")),
      limits = c(ylim_min, ylim_max),
      expand = c(0, 0),
      sec.axis = sec_axis(~ (. - ylim_min) / coeff_precip,
                          name = "降雨量 (mm)")
    ) +
    
    scale_x_date(
      name = if (show_xaxis) "日期" else "",
      date_labels = "%m-%d",
      date_breaks = "1 month",
      expand = c(0.01, 0)
    ) +
    
    theme_bw() +
    theme(
      text = element_text(family = "SimHei"),
      legend.position = if (show_xaxis) "none" else "top",  # 仅顶图显示图例
      legend.margin = margin(t = 0, r = 0, b = 0, l = 0, unit = "pt"),
      legend.text = element_text(margin = margin(r = 2, unit = "pt"), size = 10, face = "bold"),
      panel.grid = element_blank(),
      panel.border = element_rect(colour = "black", fill = NA, linewidth = 1.2),
      axis.ticks.length.y = unit(0.15, "cm"),
      axis.ticks.length.x = unit(-0.15, "cm"),
      axis.text.x = if (show_xaxis) {
        element_text(margin = margin(t = 5), size = 10, colour = "black", face = "bold")
      } else {
        element_blank()
      },
      axis.title.x = if (show_xaxis) {
        element_text(size = 10, colour = "black", face = "bold")
      } else {
        element_blank()
      },
      axis.ticks.x = if (show_xaxis) element_line() else element_blank(),
      axis.text.y.left = element_text(color = "#F08A8A", margin = margin(r = 1), size = 10, face = "bold"),
      axis.title.y.left = element_text(color = "#F08A8A", margin = margin(r = 1), size = 10, face = "bold"),
      axis.ticks.y.left = element_line(color = "#F08A8A"),
      axis.line.y.left = element_line(color = "#F08A8A", linewidth = 1),
      axis.line.y.right = element_line(color = "#4CACD6", linewidth = 1.2),
      axis.text.y.right = element_text(color = "#4CACD6", margin = margin(l = 1), size = 10, face = "bold"),
      axis.title.y.right = element_text(color = "#4CACD6", angle = 270, margin = margin(r = 1), size = 10, face = "bold"),
      axis.ticks.y.right = element_line(color = "#4CACD6")
    )
  
  # 辅助图（黄色湿度轴）
  p_axis3 <- ggplot(df_year, aes(x = Date)) +
    geom_blank() +
    scale_y_continuous(
      limits = c(ylim_min, ylim_max), expand = c(0, 0),
      sec.axis = sec_axis(~ (. - ylim_min) / coeff_hum, name = "相对湿度 (%)")
    ) +
    theme_minimal() +
    theme(
      text = element_text(family = "SimHei"),
      panel.grid = element_blank(),
      axis.text.x = element_blank(), axis.title.x = element_blank(), axis.ticks.x = element_blank(),
      axis.text.y.left = element_blank(), axis.title.y.left = element_blank(),
      axis.ticks.length.y = unit(0.15, "cm"),
      axis.text.y.right = element_text(color = "#F5B955", size = 10, face = "bold", margin = margin(r = 0,l = 5)),
      axis.title.y.right = element_text(color = "#F5B955", margin = margin(l = 3), angle = 270, size = 10, face = "bold", vjust = 0),
      axis.ticks.y.right = element_line(color = "#F5B955"),
      axis.line.y.right = element_line(color = "#F5B955", linewidth = 1)
    )
  
  # Gtable 拼接（与原代码逻辑完全一致）
  g_main   <- ggplotGrob(p)
  g_axis3  <- ggplotGrob(p_axis3)
  
  index_green_axis  <- which(g_axis3$layout$name == "axis-r")
  grob_green_axis   <- g_axis3$grobs[[index_green_axis]]
  width_green_axis  <- g_axis3$widths[g_axis3$layout[index_green_axis, "l"]]
  
  index_green_label <- which(g_axis3$layout$name == "ylab-r")
  grob_green_label  <- g_axis3$grobs[[index_green_label]]
  width_green_label <- g_axis3$widths[g_axis3$layout[index_green_label, "l"]]
  
  index_blue_label  <- which(g_main$layout$name == "ylab-r")
  col_insert_pos    <- g_main$layout[index_blue_label, "l"]
  
  width_gap <- unit(0.2, "cm")
  g_final <- gtable_add_cols(g_main,
                             widths = unit.c(width_gap, width_green_axis, width_green_label),
                             pos = col_insert_pos)
  
  index_panel <- which(g_main$layout$name == "panel")
  panel_t <- g_main$layout[index_panel, "t"]
  panel_b <- g_main$layout[index_panel, "b"]
  
  g_final <- gtable_add_grob(g_final, grob_green_axis,
                             t = panel_t, b = panel_b,
                             l = col_insert_pos + 2)
  g_final <- gtable_add_grob(g_final, grob_green_label,
                             t = panel_t, b = panel_b,
                             l = col_insert_pos + 3)
  
  return(g_final)
}

get_ellipse_poly <- function(data) {
  do.call(rbind, lapply(1:nrow(data), function(i) {
    row <- data$Row_Idx[i] # Y轴中心 (将被反转)
    col <- data$Col_Idx[i] # X轴中心
    r <- data$value[i]
    
    # 椭圆参数
    theta <- seq(0, 2 * pi, length.out = 50)
    radius <- 0.45 # 单元格内半径
    
    # 根据相关性强度压扁椭圆 (r越大越扁)
    minor_axis <- radius * (1 - abs(r))
    major_axis <- radius
    
    # 基础形状
    x_base <- major_axis * cos(theta)
    y_base <- minor_axis * sin(theta)
    
    # 旋转角度：正相关45度，负相关-45度
    angle <- ifelse(r > 0, pi/4, -pi/4)
    
    # 旋转矩阵应用
    x_rot <- x_base * cos(angle) - y_base * sin(angle)
    y_rot <- x_base * sin(angle) + y_base * cos(angle)
    y_center <- n_vars - row + 1
    data.frame(
      group_id = paste(row, col, sep = "-"),
      x = col + x_rot,        # X 轴正常 (Col 1 在左边)
      y = y_center + y_rot,   # Y 轴反转 (Row 1 在顶端)
      value = r,
      Row = data$Row[i],
      Col = data$Col[i]
    )
  }))
}

plot_pca_3d_core <- function(pca_res, var_pct, dims, colors) {
  
  point_col <- colors[1]; arrow_col <- colors[2]; ellip_col <- colors[3]
  
  # 根据 dims 提取特定的3列数据
  # 因为主函数逻辑保证了 dims 肯定是3个有效整数(如 5,6,7)，这里直接取即可
  scores_subset <- as.data.frame(pca_res$x[, dims])
  rot_subset <- as.data.frame(pca_res$rotation[, dims])
  
  colnames(scores_subset) <- c("X", "Y", "Z")
  colnames(rot_subset) <- c("X", "Y", "Z")
  
  # 下角标格式化
  format_subscripts <- function(x) {
    sapply(x, function(s) {
      if(grepl("^[A-Z][A-Z0-9]+$", s)) {
        paste0(substr(s, 1, 1), "<sub>", substr(s, 2, nchar(s)), "</sub>")
      } else { s }
    })
  }
  raw_names <- rownames(rot_subset)
  formatted_names <- format_subscripts(raw_names)
  
  # 比例缩放
  r_scores <- max(abs(scores_subset))
  r_loadings <- max(abs(rot_subset))
  scale_factor <- (r_scores / r_loadings) * 0.85
  loadings_scaled <- rot_subset * scale_factor
  
  # 椭球计算 (对应维度的 SD)
  sd_vec <- pca_res$sdev[dims]
  theta <- seq(0, 2*pi, length.out=30)
  phi <- seq(0, pi, length.out=15)
  grd <- expand.grid(theta=theta, phi=phi)
  x_ell <- 2 * sd_vec[1] * cos(grd$theta) * sin(grd$phi)
  y_ell <- 2 * sd_vec[2] * sin(grd$theta) * sin(grd$phi)
  z_ell <- 2 * sd_vec[3] * cos(grd$phi)
  
  x_mat <- matrix(x_ell, nrow=length(theta))
  y_mat <- matrix(y_ell, nrow=length(theta))
  z_mat <- matrix(z_ell, nrow=length(theta))
  
  # 轴标签
  ax_labs <- paste0("PC", dims, " (", round(var_pct[dims], 2), "%)")
  
  # 绘图配置
  axis_template <- list(showgrid = TRUE, gridcolor = "#d9d9d9", backgroundcolor = "white",
                        zerolinecolor = "#999999", titlefont = list(family = "Times New Roman", size = 14))
  
  p <- plot_ly() %>%
    add_surface(x = x_mat, y = y_mat, z = z_mat,
                opacity = 0.1, colorscale = list(c(0, 1), c(ellip_col, ellip_col)),
                showscale = FALSE, hoverinfo = "none", name = "95% Ellipse") %>%
    add_trace(data = scores_subset, x = ~X, y = ~Y, z = ~Z,
              type = 'scatter3d', mode = 'markers',
              marker = list(size = 3, color = point_col, opacity = 0.9),
              text = rownames(scores_subset), showlegend = FALSE)
  
  # 画线
  line_x <- c(); line_y <- c(); line_z <- c()
  for (i in 1:nrow(loadings_scaled)) {
    line_x <- c(line_x, 0, loadings_scaled[i, 1]*0.95, NA)
    line_y <- c(line_y, 0, loadings_scaled[i, 2]*0.95, NA)
    line_z <- c(line_z, 0, loadings_scaled[i, 3]*0.95, NA)
  }
  chang <-max(loadings_scaled[i, 1],loadings_scaled[i, 2]*0.95,loadings_scaled[i, 3]*0.95)
  p <- add_trace(p, x = line_x, y = line_y, z = line_z, type = 'scatter3d', mode = 'lines',
                 line = list(color = arrow_col, width = 4), showlegend = FALSE, hoverinfo = "none")
  
  # 画箭头头
  p <- add_trace(p, x = loadings_scaled$X, y = loadings_scaled$Y, z = loadings_scaled$Z,
                 u = loadings_scaled$X, v = loadings_scaled$Y, w = loadings_scaled$Z,
                 type = "cone", showscale = FALSE, sizemode = "absolute", sizeref = chang, anchor = "tip",
                 colorscale = list(c(0, 1), c(arrow_col, arrow_col)), name="Loadings", 
                 hoverinfo = "text", text = raw_names)
  
  # 画文字
  p <- add_trace(p, x = loadings_scaled$X * 1.15, y = loadings_scaled$Y * 1.15, z = loadings_scaled$Z * 1.15,
                 type = 'scatter3d', mode = 'text', text = formatted_names,
                 textfont = list(color = arrow_col, size = 13, family="Times New Roman"), 
                 showlegend = FALSE, hoverinfo = "none")
  
  p <- layout(p, scene = list(
    xaxis = c(axis_template, list(title = ax_labs[1])),
    yaxis = c(axis_template, list(title = ax_labs[2])),
    zaxis = c(axis_template, list(title = ax_labs[3])),
    aspectmode = "cube", camera = list(eye = list(x=1.5, y=1.5, z=1.3))
  ),
  margin = list(l=0, r=0, b=0, t=0), paper_bgcolor = "white", showlegend=FALSE)
  
  return(p)
}

spss_stepwise <- function(data, dependent_var_name, p_enter = 0.05, p_remove = 0.10) {
  
  dep_var <- dependent_var_name
  
  # 1. 明确获取所有预测变量的名称 (排除因变量)
  predictors <- setdiff(names(data), dep_var)
  
  # 2. 构建显式的最大模型公式 (例如: y ~ X1 + X2 + X3)
  # 这样 add1 函数就不会因为找不到 scope 而报错
  rhs_formula <- paste(predictors, collapse = " + ")
  scope_formula <- as.formula(paste(dep_var, "~", rhs_formula))
  
  # 3. 初始模型：只包含截距 (y ~ 1)
  current_model <- lm(as.formula(paste(dep_var, "~ 1")), data = data)
  
  steps_history <- list() 
  step_count <- 0
  
  # 记录初始状态
  step_count <- step_count + 1
  steps_history[[step_count]] <- list(model = current_model, action = "Start")
  
  changed <- TRUE
  while(changed) {
    changed <- FALSE
    
    # A. Forward Selection (Entry P <= 0.05)
    # add1 会对比当前模型和 scope 范围内的差异
    add_test <- add1(current_model, scope = scope_formula, test = "F")
    candidates <- add_test[!rownames(add_test) %in% c("<none>"), ]
    
    best_var <- NULL
    if (nrow(candidates) > 0) {
      min_p_idx <- which.min(candidates$`Pr(>F)`)
      best_p <- candidates$`Pr(>F)`[min_p_idx]
      
      if (best_p <= p_enter) {
        best_var <- rownames(candidates)[min_p_idx]
      }
    }
    
    if (!is.null(best_var)) {
      # 更新模型
      current_model <- update(current_model, paste(". ~ . +", best_var))
      changed <- TRUE
      step_count <- step_count + 1
      steps_history[[step_count]] <- list(model = current_model, action = "Entered")
    }
    
    # B. Backward Elimination (Removal P >= 0.10)
    # 只有当模型中不仅有截距时才检查
    if (length(coef(current_model)) > 1) { 
      drop_test <- drop1(current_model, test = "F")
      candidates_drop <- drop_test[!rownames(drop_test) %in% c("<none>"), ]
      
      worst_var <- NULL
      if (nrow(candidates_drop) > 0) {
        max_p_idx <- which.max(candidates_drop$`Pr(>F)`)
        worst_p <- candidates_drop$`Pr(>F)`[max_p_idx]
        
        if (worst_p >= p_remove) {
          worst_var <- rownames(candidates_drop)[max_p_idx]
        }
      }
      
      if (!is.null(worst_var)) {
        current_model <- update(current_model, paste(". ~ . -", worst_var))
        changed <- TRUE 
        steps_history[[step_count]] <- NULL
        break
      }
    }
  }
  return(steps_history)
}

  Z_minus <- apply(Z, 2, min)
  
  # 计算欧几里得距离 D+ 和 D-
  D_plus <- sqrt(rowSums(sweep(Z, 2, Z_plus, "-")^2))
  D_minus <- sqrt(rowSums(sweep(Z, 2, Z_minus, "-")^2))
  
  # 计算综合得分 C (Score)
  # Score = D- / (D+ + D-)
  C_score <- D_minus / (D_plus + D_minus)
  # 生成结果表 2：TOPSIS 排名
  topsis_result <- data.frame(
    正理想解距离 = round(D_plus, 3),
    负理想解距离 = round(D_minus, 3),
    贴合度 = round(C_score, 3)
  )
  topsis_result$排序 <- rank(-topsis_result$贴合度)

# 计算 P 值
get_model_pvalue <- function(model, y_vals) {
  tryCatch({
    if (inherits(model, "lm")) {
      f <- summary(model)$fstatistic
      if (is.null(f)) return(NA)
      p <- pf(f[1], f[2], f[3], lower.tail = FALSE)
      return(p)
    } else if (inherits(model, "nls")) {
      rss <- sum(residuals(model)^2)
      tss <- sum((y_vals - mean(y_vals))^2)
      n <- length(y_vals)
      p <- length(coef(model)) 
      df_num <- p - 1 
      df_den <- n - p
      if (df_den <= 0) return(NA)
      f_stat <- ((tss - rss) / df_num) / (rss / df_den)
      p <- pf(f_stat, df_num, df_den, lower.tail = FALSE)
      return(p)
    }
    return(NA)
  }, error = function(e) NA)
}

# 计算 NLS 的 R2 (伪 R2)
calc_r2 <- function(model, y_vals) {
  rss <- sum(residuals(model)^2)
  tss <- sum((y_vals - mean(y_vals))^2)
  return(1 - (rss / tss))
}

# 格式化数字：保留小数，去除不必要的0，处理绝对值
fmt_val <- function(x, digits = 4) {
  val <- abs(x)
  # 如果数字极小但不是0，使用科学计数法或更多小数位，否则保留指定位数
  if (val < 10^(-digits) && val > 0) {
    return(format(val, scientific = TRUE, digits = 3))
  } else {
    return(sprintf(paste0("%.", digits, "f"), val))
  }
}

formatted_stats <- final_model_stats %>%
  group_by(Trait) %>%
  arrange(desc(R2)) %>%
  slice(1) %>%
  ungroup() %>%
  mutate(
    # 处理列名中的 R2 -> R² (使用 HTML 标签)
    # 注意：这里我们处理的是数据内容，列名稍后处理
    
    # --- 处理公式列 ---
    # 1. 处理带括号的指数，例如 e^(0.5x) -> e<sup>0.5x</sup>
    Equation_Excel = str_replace_all(Equation_Excel, "\\^\\(([^)]+)\\)", "<sup>\\1</sup>"),
    
    # 2. 处理简单的数字指数，例如 x^2 -> x<sup>2</sup>
    Equation_Excel = str_replace_all(Equation_Excel, "\\^([0-9.-]+)", "<sup>\\1</sup>"),
    
    # 3. 替换乘号 * 为论文常用的 × 或省略
    Equation_Excel = str_replace_all(Equation_Excel, "\\*", "×"),
    
    # 4. 优化 R2 和 P 值的显示 (可选)
    R2 = sprintf("%.3f", R2),
    P_Value = ifelse(is.na(P_Value), "", 
                     ifelse(P_Value < 0.001, "< 0.001", sprintf("%.3f", P_Value)))
  )
